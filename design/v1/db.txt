Table users {
  id int [pk]
  email varchar [unique]
  username varchar
  firstname varchar
  lastname varchar
  createdAt date
}

Table habits {
  id int [pk]
  title varchar [not null]
  description varchar
  startDate date
  userId int [ref: > users.id]
}

Table entries {
  id int [pk]
  done date
  habitId int [ref: > habits.id]
}
